Class Mime::Type < Object

  • actionpack/lib/action_dispatch/http/mime_type.rb

Encapsulates the notion of a mime type. Can be used at render time, for example, with: 

  class PostsController < ActionController::Base
    def show
      @post = Post.find(params[:id])

      respond_to do |format|
        format.html
        format.ics { render :text => post.to_ics, :mime_type => Mime::Type["text/calendar"]  }
        format.xml { render :xml => @people.to_xml }
      end
    end
  end

107:       def parse(accept_header)
108:         if accept_header !~ /,/
109:           [Mime::Type.lookup(accept_header)]
110:         else
111:           # keep track of creation order to keep the subsequent sort stable
112:           list = []
113:           accept_header.split(/,/).each_with_index do |header, index|
114:             params, q = header.split(/;\s*q=/)
115:             if params
116:               params.strip!
117:               list << AcceptItem.new(index, params, q) unless params.empty?
118:             end
119:           end
120:           list.sort!
121: 
122:           # Take care of the broken text/xml entry by renaming or deleting it
123:           text_xml = list.index("text/xml")
124:           app_xml = list.index(Mime::XML.to_s)
125: 
126:           if text_xml && app_xml
127:             # set the q value to the max of the two
128:             list[app_xml].q = [list[text_xml].q, list[app_xml].q].max
129: 
130:             # make sure app_xml is ahead of text_xml in the list
131:             if app_xml > text_xml
132:               list[app_xml], list[text_xml] = list[text_xml], list[app_xml]
133:               app_xml, text_xml = text_xml, app_xml
134:             end
135: 
136:             # delete text_xml from the list
137:             list.delete_at(text_xml)
138: 
139:           elsif text_xml
140:             list[text_xml].name = Mime::XML.to_s
141:           end
142: 
143:           # Look for more specific XML-based types and sort them ahead of app/xml
144: 
145:           if app_xml
146:             idx = app_xml
147:             app_xml_type = list[app_xml]
148: 
149:             while(idx < list.length)
150:               type = list[idx]
151:               break if type.q < app_xml_type.q
152:               if type.name =~ /\+xml$/
153:                 list[app_xml], list[idx] = list[idx], list[app_xml]
154:                 app_xml = idx
155:               end
156:               idx += 1
157:             end
158:           end
159: 
160:           list.map! { |i| Mime::Type.lookup(i.name) }.uniq!
161:           list
162:         end
163:       end
