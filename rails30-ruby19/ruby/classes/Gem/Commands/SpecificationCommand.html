 80:       end
 81:     end

 84:       if File.exist? gem then

 86:       end

 90:       end
 91:     end
 92:

 94:       found = Gem::SpecFetcher.fetcher.fetch dep
 95: 
 96:       specs.push(*found.map { |spec,| spec })
 97:     end
 98:

100:       alert_error "Unknown gem '#{gem}'"
101:       terminate_interaction 1
102:     end
103:

105:       s = s.send field if field
106: 
107:       say case options[:format]

109:           when :marshal then Marshal.dump s

111:           end
112:
