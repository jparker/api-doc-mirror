 20:         opt.banner = "Usage: dbconsole [options] [environment]"

 22:           include_password = true
 23:         end
 24: 
 25:         opt.on("--mode [MODE]", ['html', 'list', 'line', 'column'],
 26:           "Automatically put the sqlite3 database in the specified mode (html, list, line, column).") do |mode|
 27:             options['mode'] = mode
 28:         end
 29: 
 30:         opt.on("-h", "--header") do |h|
 31:           options['header'] = h
 32:         end
 33:

 36:       end

 39:         abort "No database is configured for the environment '#{Rails.env}'"

 43:       def find_cmd(*commands)
 44:         dirs_on_path = ENV['PATH'].to_s.split(File::PATH_SEPARATOR)
 45:         commands += commands.map{|cmd| "#{cmd}.exe"} if RbConfig::CONFIG['host_os'] =~ /mswin|mingw/
 46: 
 47:         full_path_command = nil
 48:         found = commands.detect do |cmd|
 49:           dir = dirs_on_path.detect do |path|
 50:             full_path_command = File.join(path, cmd)
 51:             File.executable? full_path_command
 52:           end
 53:         end
 54:         found ? full_path_command : abort("Couldn't find database client: #{commands.join(', ')}. Check your $PATH and try again.")
 55:       end
 56: 
 57:       case config["adapter"]
 58:       when /^mysql/
 59:         args = {
 60:           'host'      => '--host',
 61:           'port'      => '--port',
 62:           'socket'    => '--socket',
 63:           'username'  => '--user',
 64:           'encoding'  => '--default-character-set'
 65:         }.map { |opt, arg| "#{arg}=#{config[opt]}" if config[opt] }.compact
 66: 
 67:         if config['password'] && include_password
 68:           args << "--password=#{config['password']}"
 69:         elsif config['password'] && !config['password'].to_s.empty?
 70:           args << "-p"
 71:         end
 72: 
 73:         args << config['database']
 74: 
 75:         exec(find_cmd('mysql', 'mysql5'), *args)
 76: 
 77:       when "postgresql"
 78:         ENV['PGUSER']     = config["username"] if config["username"]
 79:         ENV['PGHOST']     = config["host"] if config["host"]
 80:         ENV['PGPORT']     = config["port"].to_s if config["port"]
 81:         ENV['PGPASSWORD'] = config["password"].to_s if config["password"] && include_password
 82:         exec(find_cmd('psql'), config["database"])
 83: 
 84:       when "sqlite"
 85:         exec(find_cmd('sqlite'), config["database"])
 86: 
 87:       when "sqlite3"
 88:         args = []
 89: 
 90:         args << "-#{options['mode']}" if options['mode']
 91:         args << "-header" if options['header']
 92:         args << config['database']
 93: 
 94:         exec(find_cmd('sqlite3'), *args)
 95: 
 96:       when "oracle", "oracle_enhanced"
 97:         logon = ""
 98: 
 99:         if config['username']
100:           logon = config['username']
101:           logon << "/#{config['password']}" if config['password'] && include_password
102:           logon << "@#{config['database']}" if config['database']
103:         end
104: 
105:         exec(find_cmd('sqlplus'), logon)
106: 
107:       else
