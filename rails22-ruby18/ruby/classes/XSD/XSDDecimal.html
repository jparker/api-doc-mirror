250:   def screen_data_str(str)
251:     /^([+\-]?)(\d*)(?:\.(\d*)?)?$/ =~ str.to_s.strip
252:     unless Regexp.last_match
253:       raise ValueSpaceError.new("#{ type }: cannot accept '#{ str }'.")
254:     end
255:     sign = $1 || '+'
256:     int_part = $2
257:     frac_part = $3

259:     frac_part = frac_part ? frac_part.sub(/0+$/, '') : ''
260:     point = - frac_part.size

262:     # normalize

264:       sign = ''
265:     elsif sign == '-'

268:       end
269:     end
