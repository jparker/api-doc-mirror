168:   def do_write(s)
169:     @wbuffer = "" unless defined? @wbuffer
170:     @wbuffer << s
171:     @sync ||= false
172:     if @sync or @wbuffer.size > BLOCK_SIZE or idx = @wbuffer.rindex($/)
173:       remain = idx ? idx + $/.size : @wbuffer.length
174:       nwritten = 0
175:       while remain > 0

177:         begin
178:           nwrote = syswrite(str)

180:           retry
